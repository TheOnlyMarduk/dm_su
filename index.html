function loadAndDrawParking() { fetch('parken.geojson') .then(res => res.json()) .then(data => { data.features.forEach(feature => { if (!feature.geometry || feature.geometry.type !== 'Point') return; const coords = feature.geometry.coordinates; const latlng = L.latLng(coords[1], coords[0]); // Hole den Wert aus properties.adresse const adresse = feature.properties.adresse || "Parkplatz"; // Funktion zur Marker-Erstellung je nach Zoom const parkIcon = () => L.divIcon({ className: "start-goal-icon", iconAnchor: [15, 15], html: createScaledMarkerLabel("P", 30), // Großes "P" interactive: false }); // Marker hinzufügen const marker = L.marker(latlng, { icon: parkIcon() }).addTo(map); // Tooltip mit Adresse anzeigen marker.bindTooltip(adresse, { permanent: false, direction: 'top' }); // Bei Zoom: Icon neu skalieren map.on('zoomend', () => { marker.setIcon(parkIcon()); }); }); }) .catch(err => { console.error("Fehler beim Laden der parken.geojson:", err); }); }
